/// Every object-typed success value crossing into a fallible callback or
/// export must be wrapped like this -- `@boltffi/runtime`'s
/// `matchWireResult` only lets bare primitives pass through as an implicit
/// "ok"; any other object is rejected as an ambiguous `WireResult`/`WireErr`
/// shape. Wrapping unconditionally (including primitives) is always safe.
$$js.JSObject $$wireOk($$js.JSAny? value) {
  final result = $$js.JSObject();
  result.setProperty('tag'.toJS, 'ok'.toJS);
  result.setProperty('value'.toJS, value);
  return result;
}

/// Failure counterpart of [$$wireOk]. This must be *thrown*, matching
/// `matchWireResult`'s own `WireErr` handling, not returned -- a callback
/// implementation reports failure through a Dart exception even though the
/// underlying Rust method returns `Result::Err`.
///
/// Wrapped in a genuine Dart `Exception` class rather than thrown as a raw
/// `JSObject`: dart2wasm's exception machinery requires a real Dart object
/// identity to throw/catch (a bare js_interop external reference fails with
/// "Attempting to box non-Dart object"), so the JS-shaped payload travels
/// as a field instead.

  final $$js.JSObject value;
  const _WireErrException(this.value);
}

  final result = $$js.JSObject();
  result.setProperty('tag'.toJS, 'err'.toJS);

/// Awaits a JS `Promise` from calling an exported fallible Rust async
/// function. The compiled TS module's fallible functions never resolve to a
/// `{tag, value}` wrapper -- they resolve with the plain success value and
/// *reject* with a real typed `XxxException` (holding the decoded error
/// payload on its `.value`) on failure, matching ordinary JS/TS promise
/// idioms (see `target::typescript::render::function`'s `Failure::render`,
/// shared between the sync and async completion paths).

  $$js.JSPromise<$$js.JSAny?> promise,
  T Function($$js.JSAny? value) decodeSuccess,
  Object Function($$js.JSAny error) decodeError,
) async {

    final result = await promise.toDart;
    return decodeSuccess(result);
  } catch (error) {
    if (error is $$js.JSObject && error.hasProperty('value'.toJS).toDart) {
      final value = error.getProperty('value'.toJS);
      throw decodeError(value as $$js.JSAny);
    }
    rethrow;
  }
}

/// Matches the wasm/TS runtime's own `Duration` wire shape (`{secs, nanos}`,

  final micros = value.inMicroseconds;
  final secs = micros ~/ 1000000;
  final nanos = (micros % 1000000) * 1000;
  final result = $$js.JSObject();
  result.setProperty('secs'.toJS, $$dartIntToJSBigInt(secs));
  result.setProperty('nanos'.toJS, nanos.toDouble().toJS);
  return result;
}

  final nanos = (value.getProperty('nanos'.toJS) as $$js.JSNumber).toDartInt;
  return Duration(microseconds: secs * 1000000 + nanos ~/ 1000);

/// Wraps an exported async call's `Promise<T>` as a `CancelableOperation<T>`
/// so exported async calls have the identical public signature as the
/// native (dart:ffi) target, which can drive real cancellation. On web,
/// cancellation is best-effort: the wasm computation can't actually be
/// aborted mid-flight once started, so `.cancel()` here just stops the
/// caller from observing the eventual result -- same trade-off any
/// Future-based API has, just made explicit by matching the native type.
$$asyncutil.CancelableOperation<T> $$asCancelable<T>(Future<T> future) {
  return $$asyncutil.CancelableOperation<T>.fromFuture(future);
}
